#!/bin/bash
#SBATCH --job-name=zenodo_p01
#SBATCH --cpus-per-task=1
#SBATCH --mem=2g
#SBATCH --mail-type=ALL

set -euo pipefail
cd "$SLURM_SUBMIT_DIR"

ZENODO_TOKEN="72M9zlS5Ria9cEpbV9gCuCyZwdLc0SCu2rKw1xmdrEPXvnRLR0CgATCIyFYu"
API="https://zenodo.org/api"
F="nguyenlab-il15act-dataset-v1.tar.gz.part-01"

export https_proxy=${https_proxy:-"http://proxy.nih.gov:8080"}
export http_proxy=${http_proxy:-"http://proxy.nih.gov:8080"}

echo "Start: $(date)"
stat -c "%n %s" "$F" | column -t

echo "Creating draft deposition..."
DEP_JSON="$(curl -sS --fail -H "Content-Type: application/json" -H "Authorization: Bearer ${ZENODO_TOKEN}" -X POST -d '{}' "${API}/deposit/depositions")"
echo "$DEP_JSON" > dep_part2.json

DEP_ID="$(echo "$DEP_JSON" | python -c 'import json,sys; print(json.load(sys.stdin)["id"])')"
BUCKET="$(echo "$DEP_JSON" | python -c 'import json,sys; print(json.load(sys.stdin)["links"]["bucket"])')"

echo "DEP_ID=$DEP_ID"
echo "BUCKET=$BUCKET"

echo "Uploading file..."
RESP="zenodo_part01_upload_response.json"
set +e
HTTP_CODE="$(curl -sS -H "Authorization: Bearer ${ZENODO_TOKEN}" -H "Content-Type: application/octet-stream" -H "Expect:" -X PUT --upload-file "$F" "${BUCKET}/$(basename "$F")" -o "$RESP" -w "%{http_code}")"
RET=$?
set -e

echo "curl_exit=$RET http_code=$HTTP_CODE"
if [ "$RET" -ne 0 ] || [ "$HTTP_CODE" -lt 200 ] || [ "$HTTP_CODE" -ge 300 ]; then
  echo "Upload FAILED. Response saved to $RESP"
  head -c 2000 "$RESP" || true
  echo
  exit 1
fi

echo "Setting metadata..."
curl -sS --fail -H "Content-Type: application/json" -H "Authorization: Bearer ${ZENODO_TOKEN}" -X PUT -d '{"metadata":{"title":"nguyenlab-il15act dataset (v1) - Part 2","upload_type":"dataset","description":"Part 2 of 2. Contains nguyenlab-il15act-dataset-v1.tar.gz.part-01. Recombine after downloading both parts:\ncat nguyenlab-il15act-dataset-v1.tar.gz.part-* > nguyenlab-il15act-dataset-v1.tar.gz\ntar -xzf nguyenlab-il15act-dataset-v1.tar.gz","creators":[{"name":"Kanfer, Gil"}],"access_right":"open","license":"cc-by-4.0"}}' "${API}/deposit/depositions/${DEP_ID}" > /dev/null

echo "Publishing..."
curl -sS --fail -H "Authorization: Bearer ${ZENODO_TOKEN}" -X POST "${API}/deposit/depositions/${DEP_ID}/actions/publish" > /dev/null

echo "Published: https://zenodo.org/records/${DEP_ID}"
echo "End: $(date)"
